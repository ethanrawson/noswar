import os
import joblib
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression

model = Pipeline([
    ("scaler", StandardScaler()),
    ("logreg", LogisticRegression(max_iter=1000))
])

BASE_DIR = os.path.dirname(__file__)
OUT_DIR = os.path.join(BASE_DIR, "..", "results")
os.makedirs(OUT_DIR, exist_ok=True)

FEATURES = ["pct_diff","pct_ratio","gd_diff", "shots_diff", "pp_pct_diff", "pk_pct_diff"]
TRAIN_MAX_SEASON = 2018
TEST_MIN_SEASON = 2019

def main():
    df = pd.read_csv(os.path.join(OUT_DIR, "features.csv"))
    df = df.dropna(subset=FEATURES + ["home_win", "season"])

    train = df[df["season"] <= TRAIN_MAX_SEASON].copy()
    test = df[df["season"] >= TEST_MIN_SEASON].copy()

    X_train = train[FEATURES]
    y_train = train["home_win"].astype(int)

    model = LogisticRegression(max_iter=3000, C=1.0)
    model.fit(X_train, y_train)

    model_path = os.path.join(OUT_DIR, "logreg_model.joblib")
    joblib.dump(model, model_path)

    # Coefficient plot
    coef = model.coef_[0]
    coef_df = pd.DataFrame({"feature": FEATURES, "coef": coef}).sort_values("coef")

    plt.figure()
    plt.barh(coef_df["feature"], coef_df["coef"])
    plt.xlabel("Coefficient")
    plt.title("Logistic Regression Coefficients")
    out_plot = os.path.join(OUT_DIR, "feature_coefficients.png")
    plt.tight_layout()
    plt.savefig(out_plot, dpi=200)
    plt.close()

    print(f"✅ Saved model: {model_path}")
    print(f"✅ Saved plot:  {out_plot}")
    print("Coefficients:")
    for f, c in zip(FEATURES, coef):
        print(f"  {f}: {c:.4f}")
    print(f"Train rows: {len(train):,} | Test rows: {len(test):,}")

if __name__ == "__main__":
    main()
