import mysql.connector
from collections import defaultdict
import datetime as dt

DB_CONFIG = {
    "host": "localhost",
    "user": "nhl_user",
    "password": "NHLrocks101",
    "database": "nhl_model_v2",
}

CUTOFF = dt.date(2025, 12, 10)

def main():
    conn = mysql.connector.connect(**DB_CONFIG)
    cur = conn.cursor(dictionary=True)

    # Pull ALL games chronologically
    cur.execute("""
        SELECT game_id, game_date, season, team_name, is_home, won, pre_game_point_pct
        FROM team_games
        WHERE game_date IS NOT NULL
        ORDER BY season, game_date, game_id, is_home DESC
    """)

    rows = cur.fetchall()

    pts = defaultdict(int)
    gp = defaultdict(int)

    updates = []

    for r in rows:
        season = r["season"]
        team = r["team_name"]
        key = (season, team)
        game_date = r["game_date"]

        # Only compute new pct for post-cutoff games
        if game_date > CUTOFF:
            pre_pct = pts[key] / (2.0 * gp[key]) if gp[key] > 0 else None
            updates.append((pre_pct, r["game_id"], team, r["is_home"]))

        # Update standings AFTER the game
        if r["won"] is not None:
            pts[key] += 2 if int(r["won"]) == 1 else 0

        gp[key] += 1

    cur2 = conn.cursor()
    cur2.executemany("""
        UPDATE team_games
        SET pre_game_point_pct = %s
        WHERE game_id = %s
          AND team_name = %s
          AND is_home = %s
    """, updates)

    conn.commit()

    print(f"✅ Updated {cur2.rowcount} recent rows")

    cur2.close()
    cur.close()
    conn.close()

if __name__ == "__main__":
    main()
